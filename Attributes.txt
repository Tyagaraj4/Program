/* Attributes allows you to add declarative information to the program.  */

using System;
using System.Collections.Generic;
using System.IO;
using System.Runtime.Serialization;
namespace ConsoleApp1

{
    public class Program 
    {
        public static void Main()
        {
            Calculate.add(100, 100);
            Calculate.add(new List<int>() { 1,4,5,6,9});
        }
    } 
    public class Calculate
    {
        [Obsolete("What to use...!")]
        public static int add(int One, int Two)
        {
            return One + Two;
        }
        public static  int add(List<int> Nubers)
        {
            int sum = 0;
            foreach(int number in Nubers)
            {
                sum += number;
            }
            return sum;
        }
    } 

}
